# Start mapping for variant calling

## ubams
```
#!/bin/bash -e
#SBATCH --job-name=gatk_ubams
#SBATCH --output=%x.%j.out
#SBATCH --error=%x.%j.err
#SBATCH --time=00:40:00
#SBATCH --mem=32G

module load GATK/4.4.0.0-gimkl-2022a

# create temporary directory for Java so it does not fill up /tmp, and tell Java to use it
TMPDIR=/nesi/nobackup/uoo03761/GATK_tmp/
mkdir -p ${TMPDIR}
export _JAVA_OPTIONS=-Djava.io.tmpdir=${TMPDIR}

for filename in 1.fastq_files/*_1.fastq.gz
do
  base=$(basename ${filename} _1.fastq.gz)
	echo "processing ${base}"
  gatk FastqToSam -SM Overseas -F1 1.fastq_files/${base}_1.fastq.gz -F2 1.fastq_files/${base}_2.fastq.gz -O 2.ubams/${base}.unmapped.bam -PL ILLUMINA --TMP_DIR ${TMPDIR}
done
```


## Mapping

```
bwa mem -M -t ${SLURM_CPUS_PER_TASK} 1.ref_genome/Amanita_muscaria_genome_2024.fasta ../source_files/trimmed_files/${base}_1.fastq.gz ../source_files/trimmed_files/${base}_2.fastq.gz > 3.mapped_reads/${base}.full.sam
samtools view -S -b 3.mapped_reads/${base}.full.sam > 3.mapped_reads/${base}.full.bam
```

I ran this bit to remuve duplicate reads manually
```
module load SAMtools/1.22-GCC-12.3.0
for file in SRR37056118 SRR37056119 SRR37056120 SRR37056122 SRR37056123 SRR37056124 ; do samtools view -b -e '!([XA]||[SA])' 3.mapped_reads/${file}.bam > 3.mapped_reads/${file}.unique.bam ; done
```

Now the GATK pre-processing steps:
```
module load GATK/4.4.0.0-gimkl-2022a
genome=Amanita_muscaria_genome_unmasked.fasta
gatk MergeBamAlignment -ALIGNED 3.mapped_reads/${file}.unique.bam -UNMAPPED 2.ubams/${file}.unmapped.bam -O 4.preprocessing/${file}.merged.bam -R ${genome}    # Match these with ubam reads #skipping this as no extra info in ubams
gatk MarkDuplicatesSpark -I 3.mapped_reads/${file}.unique.bam -O 4.preprocessing/${file}.markeddups.bam        # Mark duplicate reads
gatk HaplotypeCaller -R ${genome} -I 4.preprocessing/${file}.markeddups.bam -O 5.hapcalling/${file}.g.vcf.gz  -ERC GVCF
```
